  {

  }

% #2: item
\cs_new:Npn \xsim_attribute_get:nn #1#2
  {
    \cs_if_exist:cTF {____xsim_attribute_ \tl_to_str:n {#1::#2} :}
      { \use:c {____xsim_attribute_ \tl_to_str:n {#1::#2} :} }
      {
        \xsim_if_debug:T
          { \msg_expandable_error:nnn {xsim} {attribute-not-set} {#1::#2} }
      }
  }
