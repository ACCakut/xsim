\XSIMmodule{solutions}{managing solutions}

\xsim_load_modules:n {tags,properties}

\bool_new:N \l__xsim_inside_solution_bool

% #1: type
% #2: id
\prg_new_protected_conditional:Npnn \xsim_solution_if_print:nn #1#2 {T,F,TF}
  {
    \bool_set_true:N \l____xsim_print_bool
    \bool_if:cF {l__xsim_#1_solution_print_bool}
      { \bool_set_false:N \l____xsim_print_bool }
    \xsim_if_tagged:nnF {#1} {#2}
      { \bool_set_false:N \l____xsim_print_bool }
    \xsim_if_in_list:nnF {print} {#2}
      { \bool_set_false:N \l____xsim_print_bool }
    \bool_if:NTF \l____xsim_print_bool
      { \prg_return_true: }
      { \prg_return_false: }
  }
\cs_generate_variant:Nn \xsim_solution_if_print:nnT {nV}

% #1: type
% #2: options
\cs_new_protected:Npn \xsim_start_solution:nn #1#2
  {
    \xsim_set_properties:nVx {#1} \g_xsim_exercise_id_tl
      { name = \xsim_get_parameter:nn {#1} {solution-name} }
    \keys_set:xn { xsim/\xsim_get_parameter:nn {#1} {solution-env} } {#2}
    \bool_set_true:N \l__xsim_inside_solution_bool
  }

\cs_new_protected:Npn \xsim_stop_solution:n #1 {}

\prg_new_conditional:Npnn \xsim_if_inside_solution: {p,T,F,TF}
  {
    \bool_if:NTF \l__xsim_inside_solution_bool
      { \prg_return_true: }
      { \prg_return_false: }
  }

% ----------------------------------------------------------------------------

% #1: type
% #2: id
\cs_new_protected:Npn \xsim_insert_solution:nn #1#2
  {
    \xsim_set_file_signature:nnn {#1} {#2} {solution}
    \file_if_exist:VT \l_xsim_file_name_tl
      {
        \group_begin:
          \xsim_start_solution:nn {#1} {}
          \tl_set:Nn \ExerciseID {#2}
          \par
          \xsim_use_template:nx
            {begin}
            { \xsim_get_parameter:nn {#1} {solution-template} }
            \file_input:V \l_xsim_file_name_tl
          \xsim_use_template:nx
            {end}
            { \xsim_get_parameter:nn {#1} {solution-template} }
          \par
        \group_end:
      }
  }
\cs_generate_variant:Nn \xsim_insert_solution:nn {nV}
  
% TODO: add options (template choice, ... )
% #1: type
\cs_new_protected:Npn \xsim_print_type_solutions:n #1
  {
    \group_begin:
      \tl_set:Nn \ExerciseType {#1}
      \xsim_use_template:nn {heading} {section*}
      \xsim_foreach_exercise:n
        { \xsim_insert_solution:nn {#1} {##1} }
    \group_end:
  }

% TODO: add options (print subheadings, sort per type,...)
\cs_new_protected:Npn \xsim_print_all_solutions:
  {
    \xsim_foreach_exercise_type:n
      { \xsim_print_type_solutions:n {##1} }
  }
